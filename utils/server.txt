"""
A zero-dependency, dual-pane Python 2 server providing:
1. Dynamic, navigable directory listings with instant downloads.
2. An interactive drag-and-drop file upload box that saves files 
directly into the active navigated folder.
"""

import os
import sys
import cgi
import urllib
import StringIO
import BaseHTTPServer
import SimpleHTTPServer

class CustomHTTPRequestHandler(SimpleHTTPServer.SimpleHTTPRequestHandler):

    def do_POST(self):
        """
        Handles multipart file uploads and writes the uploaded files 
        directly into the currently navigated local directory.
        """
        # Determine current target path from the request URL
        target_dir = self.translate_path(self.path)
        if not os.path.isdir(target_dir):
            target_dir = os.path.dirname(target_dir)

        # Parse multipart form data
        content_type = self.headers.getheader('content-type')
        if not content_type:
            content_type = self.headers.get('content-type', '')

        form = cgi.FieldStorage(
            fp=self.rfile,
            headers=self.headers,
            environ={
                'REQUEST_METHOD': 'POST',
                'CONTENT_TYPE': content_type
            }
        )

        if 'file' in form:
            file_item = form['file']
            if file_item.filename:
                # Strip leading paths to prevent directory traversal attacks
                filename = os.path.basename(file_item.filename)
                target_filepath = os.path.join(target_dir, filename)

                # Write file in chunks to keep memory usage low
                try:
                    with open(target_filepath, 'wb') as f:
                        while True:
                            chunk = file_item.file.read(65536)
                            if not chunk:
                                break
                            f.write(chunk)

                    # Respond with Success
                    self.send_response(200)
                    self.send_header('Content-Type', 'text/plain')
                    self.end_headers()
                    self.wfile.write("Upload successful")
                    return
                except IOError as e:
                    self.send_response(500)
                    self.end_headers()
                    self.wfile.write("Failed to save file: " + str(e))
                    return

        self.send_response(400)
        self.end_headers()
        self.wfile.write("Invalid form data or empty file")

    def list_directory(self, path):
        """
        Overrides the standard Python directory lister to produce a modern
        dual-pane UI with folder navigation on the left and upload box on the right.
        """
        try:
            entries = os.listdir(path)
        except os.error:
            self.send_error(404, "No permission to list directory")
            return None

        entries.sort(key=lambda a: a.lower())

        dirs_html = []
        files_html = []

        # Add parent directory link if not in root
        if self.path != '/':
            parts = self.path.rstrip('/').split('/')
            if len(parts) > 1:
                parent_path = '/'.join(parts[:-1]) + '/'
            else:
                parent_path = '/'

            dirs_html.append(
                '<li class="dir-item">'
                '<span class="icon">&#128194;</span>'
                '<a href="%s">.. (Parent Directory)</a>'
                '<span class="file-size">-</span>'
                '</li>' % urllib.quote(parent_path)
            )

        # Distinguish between directories and files
        for name in entries:
            fullname = os.path.join(path, name)
            displayname = linkname = name

            if os.path.isdir(fullname):
                displayname = name + "/"
                linkname = name + "/"
                dirs_html.append(
                    '<li class="dir-item">'
                    '<span class="icon">&#128193;</span>'
                    '<a href="%s">%s</a>'
                    '<span class="file-size">-</span>'
                    '</li>' % (urllib.quote(linkname), cgi.escape(displayname))
                )
            else:
                try:
                    size = os.path.getsize(fullname)
                    if size < 1024:
                        size_str = "%d B" % size
                    elif size < 1024*1024:
                        size_str = "%.1f KB" % (size / 1024.0)
                    else:
                        size_str = "%.1f MB" % (size / (1024.0*1024.0))
                except OSError:
                    size_str = "unknown"

                files_html.append(
                    '<li class="dir-item">'
                    '<span class="icon">&#128196;</span>'
                    '<a href="%s" download>%s</a>'
                    '<span class="file-size">%s</span>'
                    '</li>' % (urllib.quote(linkname), cgi.escape(displayname), size_str)
                )

        all_items_html = "".join(dirs_html + files_html)
        displaypath = cgi.escape(urllib.unquote(self.path))

<script>
const dropZone = document.getElementById('drop-zone');
const fileInput = document.getElementById('file-input');
const progressContainer = document.getElementById('progress-container');

const progressStatus = document.getElementById('progress-status');

function handleFiles(files) {
if (files.length === 0) return;

progressContainer.style.display = 'block';
progressFill.style.width = '0%';
progressStatus.textContent = 'Uploading 0/' + files.length + '...';

uploadFilesSequential(Array.from(files));
}

async function uploadFilesSequential(files) {
for (let i = 0; i < files.length; i++) {
const file = files[i];
progressStatus.textContent = `Uploading ${i+1}/${files.length}: ${file.name}...`;

const formData = new FormData();
formData.append('file', file);

try {
await new Promise((resolve, reject) => {
const xhr = new XMLHttpRequest();
// POST back to current route so it drops in the browsed directory
xhr.open('POST', window.location.pathname, true);

xhr.upload.addEventListener('progress', (e) => {
if (e.lengthComputable) {
const currentFileProgress = e.loaded / e.total;
const overallPercent = (((i + currentFileProgress) / files.length) * 100).toFixed(1);
progressFill.style.width = overallPercent + '%';
}
});

xhr.onload = function() {
if (xhr.status >= 200 && xhr.status < 300) {
resolve();
} else {
reject(new Error('Upload failed with status ' + xhr.status));
}
};

xhr.onerror = function() {
reject(new Error('Network error'));
};

xhr.send(formData);
});
} catch (error) {
alert('Error uploading ' + file.name + ': ' + error.message);
}
}

progressStatus.textContent = 'Upload complete! Refreshing page...';
progressFill.style.width = '100%';
setTimeout(() => {
window.location.reload();
}, 1200);
}
</script>

        # Perform replacement safely without escaping curly braces inside CSS/JS

        # Prepare returning content
        f = StringIO.StringIO()
        f.write(encoded_html)
        f.seek(0)

        self.send_response(200)
        self.send_header("Content-Type", "text/html; charset=utf-8")
        self.send_header("Content-Length", str(len(encoded_html)))
        self.end_headers()
        return f

if __name__ == '__main__':
